# 高精度算法

$len(A) <= 1e6$ （ 是位数而不是数值 ）

### 大整数的存储：

###### 	把大整数的每一位取到数组里面去,一般数组的第0位存储个位。因为计算的时候可能会有进位，要在高位上补上一个数，在数组的末位添加一个数字比较方便。

运算的过程实际上是模拟人工运算的过程。每一次算一位的时候，实际上算得是三个数相加 

将数字字符减去一个0字符，就可以把它变成真实的整数，字符转数值

## 高精度减法

Ai - Bi 若大于等于 0 直接减 如果小于0 就跟前面借一位

则是 Ai - Bi - t 。t 就是表示借位，上一位借过位就是1，反之就是0

A -B 如果 A 大于等于 B， 就是 A - B,反之就是 - ( B - A )，减少了一些边界情况。

```C++
#include<iostream>
#include<vector>

using namespace std;

bool cmp(vector<int> &A, vector<int> &B)
{
    if(A.size() != B.size()) return A.size() > B.size();
    for (int i = A.size() - 1; i >= 0; i --)
    {
        if(A[i] != B[i])
        {
            return A[i] > B[i];
        }
    }
    return true;
}

vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;

    int t = 0;
    for (int i = 0; i < A.size(); i ++)
    {
        t = A[i] - t;
        if(i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        //综合考虑了要借位和不用借位的情况
        if( t < 0 ) t = 1;
        //要借位的话 t 就设置成 1
        else t = 0;
    } 
    //去掉前导0
    while(C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}
int main()
{   
    string a,b;
    cin >> a >> b;
    vector<int> A, B;
    for(int i = a.size() - 1; i >= 0; i --) A.push_back(a[i] - '0');
    for(int i = b.size() - 1; i >= 0; i --) B.push_back(b[i] - '0');

    if(cmp(A, B))
    {
        auto C = sub(A, B);
        
        for(int i = C.size() - 1; i >= 0; i --) printf("%d", C[i]);
    }
    else
    {
        auto C = sub(B, A);
        
        printf("-");
        for(int i = C.size() - 1; i >= 0; i --) printf("%d", C[i]);
    }
    return 0;
}
```



## 高精度乘法

$A5~A4~A3~A2~A1~A0~\times~b$ 个位就是 $A0 \times b \% 10$ ，

需要进的位就是 $A0 \times b / 10$。第二位就应该是 $( A1 \times b + t ) \% 10$

进的位就是 $( A1 \times b + t )  / 10$

```c++
#include<iostream>
#include<vector>

using namespace std;

vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;

    int t = 0; //用 t 来表示进位
    for(int i = 0; i < A.size() || t; i ++ )
    //如果 A 没循环完或者 t 还有进位，就继续做
    {
        if(i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    return C;
}

int main()
{
    string a;
    int b;
    cin >> a >> b;
    vector<int> A;
    for(int i = a.size() - 1; i >=0; i--)
    {
        A.push_back(a[i] - '0');
    }
    
    auto C = mul(A, b);

    for(int i = C.size() - 1; i >= 0; i --)
    {
        printf("%d", C[i]);
    }
    return 0;
}
```

## 高精度除法

```c++
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

vector<int> div(vector<int>&A, int b, int &r)
{
    vector<int> C;

    r=0;
    for(int i = A.size() - 1; i >= 0; i -- )
    {
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }

    reverse(C.begin(), C.end());
    //颠倒，与大整数乘法相对应
    while(C.size() > 1 && C.back() == 0) C.pop_back();
    //去掉前导0
    return C;

}
int main()
{
    string a;
    int b;
    cin >> a >> b;
    vector<int> A;
    for(int i = a.size() - 1; i >= 0; i -- )
    {
        A.push_back(a[i] - '0');
    }
    int r;
    auto C = div(A, b, r);
    for(int i = C.size() - 1; i >= 0; i -- )
    {
        printf("%d", C[i]);
    }
    printf("\n");
    cout << r << endl;
    return 0;
}
```

```c++
void solve()
{
    int n, m;
    cin >> n >> m;
    memset(a, 0, sizeof a);
    a[1] = 1;
    int w = 1;
    for(int i = 2; i <= n; i ++)
    {
        int k = 0;
        int j;
        for(j = 1; j <= w; j ++)
        {
            a[j] = a[j] * i + k;
            k = a[j] / 10;
            a[j] %= 10;
        }
        while(k)
        {
            a[j] = k % 10;
            k /= 10;
            j ++;
        }
        w = j - 1;
    }

    long long sum = 0;
    for(int i = 1; i <= w; i ++)
        if(a[i] == m)
            sum ++;
        
    cout << sum << endl;
}
```
